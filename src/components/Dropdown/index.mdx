import { Playground, Props } from 'docz'
import { useState } from 'react'
import Dropdown from './index'
import { containerWithoutHeight } from '../../styles/index'

# Dropdown
Componente de menu suspenso e alternável, permitindo que o usuário escolha ou preencha um valor a partir de uma lista predefinida.

<Playground>
    {() => {
        const [selection, setSelection] = useState('')
        return (
            <div style={{height: 100, ...containerWithoutHeight}}>
                <Dropdown
                    readOnly={false}
                    value={selection}
                    onChange={({ target: { value } }) => setSelection(value)}
                    list={['Apples', 'Bananas', 'Oranges', 'Melons', 'Berries']}
                    placeholder='Escolha uma opção'
                    onChangeKeyboard={element => element ? setSelection(element.value) : null }
                />
            </div>
        )
    }}
</Playground>

<Props of={Dropdown} />
