import InputText from './index'
import { containerWithoutHeight } from '../../styles/index'

# InputText
Componente padrão utilizado para a entrada de textos nas aplicações. Não depende de nenhum outro componente para seu funcionamento e pode ser utilizado na composição de outros componentes, à exemplo o componente **Form**.

<Playground>
  {() => {
    const [inputValue, setInputValue] = useState('')
    const inputRef = useRef(null);
    const onChange = ({ target: { value } }) => setInputValue(value)
    const placeholder = 'Digite aqui'
    const name = 'input'
    const inputProps = { value: inputValue, onChange, placeholder, name, ref: inputRef }
    
    return (
        <div style={{height: 100, ...containerWithoutHeight}}>
            <p>{inputValue}</p>
            <InputText {...inputProps} />
        </div>
    )
  }}
</Playground>
