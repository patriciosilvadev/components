# InputMoney
Componente padrão utilizado para a entrada de texto com a formatação para dinheiro. Não depende de nenhum outro componente para seu funcionamento e pode ser utilizado na composição de outros componentes, à exemplo o **Form**.

<Playground>
    {() => {
        const [inputValue, setInputValue] = useState('')
        const [submitting, setSubmitting] = useState(false)
        const inputRef = useRef(null)
        const name = 'input'
        const inputProps = { value: inputValue, submitting, name, ref: inputRef, setValue: setInputValue }
        return (
            <div style={{...containerDocz, height: '180px'}}>
                <p>&nbsp;{inputValue}</p>
                <InputMoney {...inputProps} />
                <div style={{ display: 'grid', marginTop: '40px' }} onClick={() => setSubmitting(!submitting)}><Button type='submit' cta='Disable input' /></div>
            </div>
        )
    }}
</Playground>
